func main {

%main.entry:
    call _init 
    $a = move 1
    $b = move 2
    $c = move 0
    $t = sne $a 1
    br $t %if_true %if_false

%if_false:
    $c = move $b
    jump %if_after

%if_true:
    $c = move $a
    jump %if_after

%if_after:
    ret $c

}

func _init {

%_init.entry:
    ret 0

}
